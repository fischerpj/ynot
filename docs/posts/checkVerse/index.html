import { html } from "@observablehq/stdlib";

async function fetchPassage(param) {
  const url = `https://hall.pjafischer.workers.dev/passage/?param=${param}`;
  
  try {
    const response = await fetch(url);
    if (!response.ok) {
      throw new Error(`HTTP error! Status: ${response.status}`);
    }
    return await response.json();
  } catch (error) {
    console.error("Error fetching data:", error);
    return { error: "Failed to fetch data" };
  }
}

  const [bc,verse,edition="SG21"] = input.split(/[:!]/);

  const valid_edition = editions.has(edition) ? edition : ""
  // strip trailing digits
