checkVerse

checks validity of BibleRef THEN fetch&display (with observableJS).

import { html } from "@observablehq/stdlib";

//======================================================
//===== INPUT 

// Create interactive INPUT field using viewof
viewof param = Inputs.text({ 
  label: "Search", 
  value: "gen1:1!KJV"});

// btn btn-primary btn-success
viewof addButton = Inputs.button("Add");
viewof resetButton = Inputs.button("Reset");

html`<div class="row container">${viewof param} ${viewof addButton}  ${viewof resetButton}</div>`
bcve = (() => {
  const editions = new Set(['KJV','SG21',"NGU-DE"]);
  
  // Split by multiple separators: comma, semicolon, and pipe
  const [bc,verse,edition="SG21"] = param.split(/[:!]/);
  // emptied edition is doomed invalid 
  const valid_edition = editions.has(edition) ? edition : ""
  const book = bc.split(/(\d+)$/);

  const result = {
    "is_valid": ( 
        verse === "" || 
        valid_edition === ""
        ) ? false : true,
    "edition": valid_edition,
    "verse": verse,
    "bc": bc,
    "book": book[0],
    "input": param
  }
  
  return result
})();

//======================================================
//===== FETCH & display

// Content reactive to input
content = await (async () => {
  // Craft the URL dynamically
  const url = `https://hall.pjafischer.workers.dev/passage/?param=${encodeURIComponent(param)}`;
  if ( bcve.is_valid ) {
      console.log(url);
      const response = await fetch(url);
      const text = await response.json();
      return text;
  } else {
    return bcve.is_valid
  }
})();

html`<div>${content.htmlraw}</div>`;
