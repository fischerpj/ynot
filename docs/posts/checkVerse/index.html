  const regx = /^[^!]*/;
  const regx1 = /^[^:]*/;
  const no_edition = userref.match(regx).toString();
  const no_verse = no_edition.match(regx1).toString();
  
  const result = {
    "is_valid": ( 
        verse === "" || 
        valid_edition === ""
        ) ? false : true,
    "edition": valid_edition,
    "verse": verse,
    "bc": bc,
    "input": userref
  }
  
  return result
})(param);
