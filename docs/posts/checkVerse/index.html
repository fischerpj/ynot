//======================================================
//===== CHECK 
function bcve_(input) {
  const editions = new Set(['KJV','SG21',"NGU-DE"]);
  
  // Split by multiple separators: comma, semicolon, and pipe
  const [bc,verse,edition="SG21"] = input.split(/[:!]/);
  // emptied edition is doomed invalid 
  const valid_edition = editions.has(edition) ? edition : ""
  const book = bc.split(/(\d+)$/);

  return {
    "is_valid": ( 
        verse === "" || 
        valid_edition === ""
        ) ? false : true,
    "edition": valid_edition,
    "verse": verse,
    "bc": bc,
    "book": book[0],
    "input": param
  }
};

//======================================================
//===== FETCH & display

// Content reactive to input
content = await (async () => {
  const mysplit = bcve_(param);
  // Craft the URL dynamically
  const url = `https://hall.pjafischer.workers.dev/passage/?param=${encodeURIComponent(param)}`;
  if ( mysplit.is_valid ) {
      console.log(url);
      const response = await fetch(url);
      const text = await response.json();
      return text;
  } else {
    return mysplit.is_valid
  }
})();
